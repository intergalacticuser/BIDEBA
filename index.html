  <div class="container">

    <h1 class="counter">Visitors: <span id="visitor-count">0</span></h1>

    <canvas id="falling-letters"></canvas>

  </div>

  <script>

    // Visitor Counter Logic

    let visitorCount = 0;


    function updateCounter() {

      const counterElement = document.getElementById('visitor-count');

      visitorCount++;

      counterElement.textContent = visitorCount;

    }


    // Simulate visitor count increment every 5 seconds

    setInterval(updateCounter, 5000);


    // Falling Letters Animation

    const canvas = document.getElementById('falling-letters');

    const ctx = canvas.getContext('2d');


    canvas.width = window.innerWidth;

    canvas.height = window.innerHeight;


    window.addEventListener('resize', () => {

      canvas.width = window.innerWidth;

      canvas.height = window.innerHeight;

    });


    const letters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';

    const fontSize = 20;

    const columns = canvas.width / fontSize;


    const drops = [];

    for (let i = 0; i < columns; i++) {

      drops[i] = 1;

    }


    function drawFallingLetters() {

      ctx.fillStyle = 'rgba(0, 0, 0, 0.1)';

      ctx.fillRect(0, 0, canvas.width, canvas.height);


      ctx.fillStyle = '#0f0'; // Green color for letters

      ctx.font = `${fontSize}px Arial`;


      for (let i = 0; i < drops.length; i++) {

        const text = letters[Math.floor(Math.random() * letters.length)];

        ctx.fillText(text, i * fontSize, drops[i] * fontSize);


        if (drops[i] * fontSize > canvas.height || Math.random() > 0.95) {

          drops[i] = 0;

        }

        drops[i]++;

      }

    }


    setInterval(drawFallingLetters, 30); // Update animation every 30ms

  </script>
